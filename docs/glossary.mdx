# Glossary

This page contains definitions for common terms used throughout the vCluster documentation.

{Object.entries(glossaryData)
  .sort((a, b) => a[1].term.localeCompare(b[1].term))
  .map(([key, entry]) => (
    <div key={key} id={key} style={{ marginBottom: '2rem' }}>
      <h3>{entry.term}</h3>
      <p>{entry.definition}</p>
      
      {entry.related && (
        <p><strong>Related terms:</strong> {entry.related.map((term, index) => (
          <span key={term}>
            {index > 0 && ', '}
            <a href={`#${term}`}>{glossaryData[term]?.term || term}</a>
          </span>
        ))}</p>
      )}
    </div>
  ))
}
